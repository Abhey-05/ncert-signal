\noindent \textbf{3} \quad A 44 mH inductor is connected to 220 V, 50 Hz ac supply. Determine
the rms value of the current in the circuit.\\
\solution
\fi


\begin{figure}[ht]
  \centering
      \input{ncert-physics/12/7/3/figs/ckt1}
  \caption{Circuit-1}
\end{figure}

\begin{align}
    V &= I \brak{j \omega L} \\
    I &= \frac{V}{j \omega L} A \\
    I &= \frac{220 \sqrt{2}}{j \brak{314} \brak{44 \text{x} 10^{-3}}}A\\
    I &= \frac{22.52}{j}A
\end{align}


\begin{figure}[ht]
  \centering

      \input{ncert-physics/12/7/3/figs/ckt2}

  \caption{Circuit-2}
\end{figure}

\begin{align}
    I_{rms} = \frac{I}{\sqrt{2}}A\\
    I_{rms} = \frac{15.92}{j}A\\
    \abs{I_{rms}} = 15.92 A
\end{align}
